wget https://cloud-images.ubuntu.com/focal/current/focal-server-cloudimg-amd64.img

export STORAGE_POOL="local-lvm"
export VM_ID="10000"
export VM_NAME="ubuntu-20.04-cloudimg"
export VM_NET="vmbr1"
export IMG="focal-server-cloudimg-amd64.img"

# qm destroy 10000 --purge

# Customize image before import
virt-customize -a "$IMG" \
  --install cloud-init,qemu-guest-agent \
  --run-command 'systemctl enable qemu-guest-agent' \
  --run-command 'cloud-init clean' \
  --truncate /etc/machine-id

# Create Proxmox VM image from Ubuntu Cloud Image
qm create $VM_ID --memory 1024 --cores 1 --net0 virtio,bridge=$VM_NET
qm importdisk $VM_ID "$IMG" $STORAGE_POOL
qm set $VM_ID --scsihw virtio-scsi-pci --scsi0 $STORAGE_POOL:vm-$VM_ID-disk-0
qm set $VM_ID --agent enabled=1,fstrim_cloned_disks=1
qm set $VM_ID --name $VM_NAME

# Create Cloud-Init Disk and configure boot
qm set $VM_ID --ide2 $STORAGE_POOL:cloudinit
qm set $VM_ID --boot c --bootdisk scsi0
qm set $VM_ID --serial0 socket --vga serial0

# Optional but useful
qm set $VM_ID --ostype l26
qm set $VM_ID --tablet 0

# Convert to template
qm template $VM_ID
